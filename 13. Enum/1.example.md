enum:
give names to a set of constants
names must be unique

		#include <iostream>
		enum Status{
		    Pending,    //by default position starts from zero
		    Approved,
		    Cancelled
		};

		enum Log{
		    WARNING = 1,
		    DEBUG,
		    FATAL
		};

		int main(){    
		    Status s = Pending;
		    s = Approved;    
		    std::cout << "Approved : "<< s << std::endl;    //1
		    Log log = Log::DEBUG;
		    std::cout << "Debug :" << log << std::endl;     //2
		    return 0;
		}
