#### while loop

            #include <iostream>
            using namespace std;
            int main()
            {
                int number = 50;
                while(number > 1){
                    cout << number << endl;
                    number--;
                }
                return 0;
            }


#### do while loop


          #include <iostream>
          using namespace std;
          int main()
          {
              int number = 50;
              do{
                  cout << number << endl;
                  number--;
              }while(number > 1);
              return 0;
          }

#### continue : ex ; printing only even numbers

                    {
                        //print only even numbers
