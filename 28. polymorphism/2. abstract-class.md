overriding pure virutal function/abstract method in subclass :  


          #include <iostream>  
          using namespace std;  
          class A{
            public:
                virtual void method() = 0;
          };

          //default inheritance is private , so making it public
          class B : public A{
            //overriding pure virtual function in subclass
            public:
              void method(){
                cout << "B method" << endl;
              }
          };

          int main(){  
              A *a = new B(); // no error
              return 0;  
          }
