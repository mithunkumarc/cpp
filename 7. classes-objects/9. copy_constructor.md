                    #include <iostream>  

                    using namespace std;  

                    class Demo  
                    {  
                        int a;  
                        int b;  
                        int *p;  
                        public:  
                        Demo()  
                        {  
                            p=new int;  
                        }  
                        void setdata(int x,int y,int z)  
                        {  
                            a=x;  
                            b=y;  
                            *p=z;  
                        }  
                        void showdata()  
                        {  
                            std::cout << "value of a is : " <<a << "," << &a << std::endl;  
                            std::cout << "value of b is : " <<b << "," << &b <<  std::endl;  
                            std::cout << "value of *p is : " << *p << "," << p << std::endl;  //same memory location
                        }  
                    };  
                    int main()  
                    {  
                      Demo d1;  
                      d1.setdata(4,5,7);  
                      Demo d2 = d1;  
                      d2.showdata();  
                      d1.showdata();
                        return 0;  
                    }
