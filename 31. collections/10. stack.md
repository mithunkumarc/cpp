Stacks are a type of container adaptors with LIFO(Last In First Out) type of working, 
where a new element is added at one end and (top) an element is removed from that end only.


example : 

              #include<iostream>
              #include<stack>
              void showStack(std::stack<int> s){
                  //s.empty() : returns 1 or 0
                  //1 - true , 0 - false
                  while(!s.empty()){
                      std::cout << "\t" << s.top();
                      s.pop();
                  }
                  std::cout << std::endl;
              }

              int main(){
                  std::stack<int> s;
                  s.push(10);
                  s.push(30);
                  s.push(23);
                  s.push(77);
                  s.push(57);
                  s.push(84);
                  std::cout << "stack" << std::endl;
                  showStack(s);
                  std::cout << "size() : " << s.size() << std::endl;
                  std::cout << "top() : " << s.top() << std::endl;
                  std::cout << "pop() : " << std::endl;
                  s.pop();
                  showStack(s);
                  return 0;
              }


              output : 

              stack
                      84      57      77      23      30      10
              size() : 6
              top() : 84
              pop() : removing first element
                      57      77      23      30      10
