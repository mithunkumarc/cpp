#### swap value of two numbers using pointers 

              #include <iostream>
              #include <string>
              using namespace std;

              void swap(int *n1,int *n2){
                  int temp = *n1;
                  *n1 = *n2;
                  *n2 = temp;
              }

              int main ()
              {
                  int a = 10, b = 20;
                swap(&a, &b);
                cout << "a : " << a << ", b : " << b << endl;
                return 0;
              }
