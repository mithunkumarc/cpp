#include <iostream>
auto glambda = [](auto a, auto&& b){
    std::cout << b << std::endl;
    return a < b;
};


int main(){
    std::cout << glambda(10,20) << std::endl;
}


// && : 
// It is called an rvalue reference, and it is new in C++11. It binds to temporaries without making a copy.
