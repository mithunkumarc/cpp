#### decrement counter without using loop

          #include <iostream>
          using namespace std;
          int decrementFunction(int number){
            if(number == 1)
                return number;
            number--;
            cout << number << endl;
            return decrementFunction(number);

          }

          int main()
          {
            decrementFunction(10);
            return 0;
          }



#### factorial

            #include <iostream>
            using namespace std;
            int factorial(int number){
                if(number == 1)
                    return 1;
                return number * factorial(number-1);
            }
            int main(){
                cout << factorial(5);
                return 0;
            }
