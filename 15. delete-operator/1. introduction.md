##### Release memory pointed by pointer-variable
        syntax : delete pointer-variable;  


    A* a = new A();
    delete a; //works, stored on heap
    
    // deleting memory allocated for array
    int* set = new int[100];     
    delete [] set;  
    
    
    A aa; (stored on stack)
    delete aa; //error
